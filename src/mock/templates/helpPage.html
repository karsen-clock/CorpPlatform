ConfigParameter 

--下面是基本配置，一般只用这种就可以
[
{
 "request" :
 {
   "uri" : "/foo/h",--在这里填写接口的URI
	 "text":
	  {
	  "startsWith":"" --在这里填写请求匹配项，支持多种参数(endwith contains)
	  }
    },
"Response" :--指定响应参数
{
	"file":"test.response"--在这里填写你指定响应报文保存的文件名(文件将在下一步上传，文件必须以.response结尾)
}
}
]
